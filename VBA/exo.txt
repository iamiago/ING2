Public Function setDay(Day As Integer) As String
Dim Msg As String
Select Case Day
    Case Is = 0
        Msg = "Dimanche"
    Case Is = 1
        Msg = "Lundi"
    Case Is = 2
        Msg = "Mardi"
    Case Is = 3
        Msg = "Mercredi"
    Case Is = 4
        Msg = "Jeudi"
    Case Is = 5
        Msg = "Vendredi"
    Case Is = 6
        Msg = "Samedi"
End Select
setDay = Msg
End Function
